       def get_ancestors(node_id: str, parent_map: dict[str, list[str]]) -> set[str]:
           """BFS traversal collecting all ancestor node IDs."""
           visited: set[str] = set()
           queue = list(parent_map.get(node_id, []))
           while queue:
               current = queue.pop(0)
               if current in visited:
                   continue
               visited.add(current)
               queue.extend(parent_map.get(current, []))
           return visited

       def write_dag_node(self, node: DAGNodeSnapshot) -> ArtifactRecord:
           path = self._root / "runs" / node.run_id / "dag" / "nodes" / f"{node.node_id}.json"
           self._write_model(path, node)
           return ArtifactRecord(artifact_id=f"dag-node:{node.node_id}", storage_uri=str(path), media_type="application/json")

       def load_dag_node(self, node_id: str) -> DAGNodeSnapshot | None:
           # Search across all run directories for the node
           for run_dir in sorted((self._root / "runs").glob("*/dag/nodes")):
               path = run_dir / f"{node_id}.json"
               result = self._read_model(path, DAGNodeSnapshot)
               if result is not None:
                   return result
           return None

       def list_dag_nodes(self, run_id: str) -> list[DAGNodeSnapshot]:
           nodes_dir = self._root / "runs" / run_id / "dag" / "nodes"
           if not nodes_dir.exists():
               return []
           return [DAGNodeSnapshot.model_validate(json.loads(p.read_text())) for p in sorted(nodes_dir.glob("*.json"))]

       def write_dag_edge(self, edge: DAGEdgeSnapshot) -> ArtifactRecord:
           # Need run_id -- derive from source node
           source_node = self.load_dag_node(edge.source_node_id)
           run_id = source_node.run_id if source_node else "unknown"
           path = self._root / "runs" / run_id / "dag" / "edges" / f"{edge.source_node_id}-{edge.target_node_id}.json"

           return ArtifactRecord(artifact_id=f"dag-edge:{edge.source_node_id}-{edge.target_node_id}", storage_uri=str(path), media_type="application/json")
