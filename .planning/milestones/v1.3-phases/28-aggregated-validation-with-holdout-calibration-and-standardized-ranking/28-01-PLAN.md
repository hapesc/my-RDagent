   ```python
   """Holdout validation protocol abstractions for Phase 28 aggregated validation."""
   from __future__ import annotations
   from typing import Protocol
   from pydantic import BaseModel, ConfigDict, Field

   class FoldSpec(BaseModel):
       """Opaque fold descriptor produced by HoldoutSplitPort."""
       model_config = ConfigDict(extra="forbid", frozen=True)
       fold_index: int = Field(ge=0)
       train_ref: str = Field(min_length=1)
       holdout_ref: str = Field(min_length=1)

   class HoldoutSplitPort(Protocol):
       """Called once per finalization to produce K fold specs."""
       def split(self, *, run_id: str) -> list[FoldSpec]: ...

   class EvaluationPort(Protocol):
       """Called once per candidate per fold to evaluate holdout performance."""
       def evaluate(self, *, candidate_node_id: str, fold: FoldSpec) -> float: ...

   class StratifiedKFoldSplitter:
       """Default K-fold splitter producing opaque partition references."""
       def __init__(self, k: int = 5) -> None:
           self._k = k
       def split(self, *, run_id: str) -> list[FoldSpec]:
           return [
               FoldSpec(fold_index=i, train_ref=f"{run_id}-train-fold-{i}", holdout_ref=f"{run_id}-holdout-fold-{i}")
               for i in range(self._k)
           ]

   class StubHoldoutSplitPort:
       """Deterministic stub for tests."""
       def __init__(self, k: int = 5) -> None:
           self._k = k
       def split(self, *, run_id: str) -> list[FoldSpec]:
           return [
               FoldSpec(fold_index=i, train_ref=f"train-fold-{i}", holdout_ref=f"holdout-fold-{i}")
               for i in range(self._k)
           ]

   class StubEvaluationPort:
       """Returns a fixed score per candidate for deterministic testing."""
       def __init__(self, scores: dict[str, float] | None = None) -> None:
           self._scores = scores or {}
       def evaluate(self, *, candidate_node_id: str, fold: FoldSpec) -> float:
           return self._scores.get(candidate_node_id, 0.5)

   __all__ = [
       "EvaluationPort",
       "FoldSpec",
       "HoldoutSplitPort",
       "StratifiedKFoldSplitter",
       "StubEvaluationPort",
       "StubHoldoutSplitPort",
   ]
   ```

2. Create `v3/algorithms/holdout.py`:
   ```python
   """Pure holdout validation algorithms for Phase 28."""
   from __future__ import annotations
   import statistics
   from v3.contracts.exploration import DAGNodeSnapshot, DAGEdgeSnapshot, EdgeType

   def rank_candidates(
       candidate_scores: dict[str, list[float]],
   ) -> list[tuple[str, float, float]]:
       """Rank by mean holdout score (desc), tiebreak by std (asc).
       Returns list of (node_id, mean, std) tuples."""
       if not candidate_scores:
           return []
       entries = []
       for node_id, fold_scores in candidate_scores.items():
           mean_score = statistics.mean(fold_scores)
           std_score = statistics.stdev(fold_scores) if len(fold_scores) > 1 else 0.0
           entries.append((node_id, mean_score, std_score))
       entries.sort(key=lambda e: (-e[1], e[2]))
       return entries

   def filter_by_quality_threshold(
       candidates: list[DAGNodeSnapshot],
   ) -> list[DAGNodeSnapshot]:
       """Filter candidates below median validation_score."""
       if len(candidates) <= 1:
           return list(candidates)
       scores = [c.node_metrics.validation_score for c in candidates]
       threshold = statistics.median(scores)
       return [c for c in candidates if c.node_metrics.validation_score >= threshold]

   def collect_candidate_ids(
       all_nodes: list[DAGNodeSnapshot],
       all_edges: list[DAGEdgeSnapshot],
       frontier_ids: set[str],
   ) -> set[str]:
       """Collect frontier + MERGED target node IDs, deduplicated."""
       merged_target_ids = {
           edge.target_node_id
           for edge in all_edges
           if edge.edge_type == EdgeType.MERGED
       }
       return frontier_ids | merged_target_ids

   __all__ = ["collect_candidate_ids", "filter_by_quality_threshold", "rank_candidates"]
   ```
