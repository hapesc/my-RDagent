   ```python
   def build_finalization_guidance(
       *,
       submission: FinalSubmissionSnapshot,
   ) -> OperatorGuidance:
       """Build operator guidance for Phase 28 finalization results."""
       ranking_lines = []
       for entry in submission.ranked_candidates:
           ranking_lines.append(
               f"  #{entry.rank}: node={entry.node_id} branch={entry.branch_id} "
               f"holdout_mean={entry.holdout_mean:.4f} holdout_std={entry.holdout_std:.4f}"
           )
       ranking_table = "\n".join(ranking_lines) if ranking_lines else "  (no candidates evaluated)"
       ancestry_text = " -> ".join(submission.ancestry_chain) if submission.ancestry_chain else "(root node)"

       return OperatorGuidance(
           recommended_next_skill="rd-evaluate",
           current_state=(
               f"Current state: finalization complete for run {submission.run_id}. "
               f"Winner: node {submission.winner_node_id} from branch {submission.winner_branch_id} "
               f"with holdout_mean={submission.holdout_mean:.4f} and holdout_std={submission.holdout_std:.4f}."
           ),
           routing_reason=(
               f"Reason: holdout evaluation ranked {len(submission.ranked_candidates)} candidate(s). "
               f"The winner was selected by highest mean holdout score across K folds.\n"
               f"Ranking:\n{ranking_table}"
           ),
           exact_next_action=(
               "Next action: review the winning solution and prepare for submission. "
               "Use rd-evaluate to inspect the final artifacts."
           ),
           next_step_detail=(
               f"Winner ancestry: {ancestry_text}\n"
               f"Submission ID: {submission.submission_id}"
           ),
       )
   ```
