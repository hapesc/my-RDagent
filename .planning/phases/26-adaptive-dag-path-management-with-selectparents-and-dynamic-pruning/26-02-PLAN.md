    ```python
    from __future__ import annotations
    from uuid import uuid4
    from v3.algorithms.dag import get_ancestors, get_descendants, get_frontier, get_depth
    from v3.contracts.exploration import DAGEdgeSnapshot, DAGNodeSnapshot, EdgeType, NodeMetrics
    from v3.ports.state_store import StateStorePort

    class DAGService:
        """Graph topology layer over the branch lifecycle model.

        Branch tracks execution state (active/superseded/completed).
        DAG tracks topology (parent-child, depth, path metrics).
        """

        def __init__(self, state_store: StateStorePort) -> None:
            self._state_store = state_store

        def create_node(
            self,
            *,
            run_id: str,
            branch_id: str,
            parent_node_ids: list[str] | None = None,
            node_metrics: NodeMetrics | None = None,
        ) -> DAGNodeSnapshot:
            parent_ids = parent_node_ids or []
            node_id = f"dag-node-{uuid4().hex[:12]}"

            # Validate parents exist
            for pid in parent_ids:
                if self._state_store.load_dag_node(pid) is None:
                    raise KeyError(f"parent node not found: {pid}")

            # Build parent_map for depth computation
            all_nodes = self._state_store.list_dag_nodes(run_id)
            parent_map = {n.node_id: list(n.parent_node_ids) for n in all_nodes}
            parent_map[node_id] = parent_ids

            # Validate no cycles (new node's ancestors must not include node_id)
            ancestors = get_ancestors(node_id, parent_map)
            if node_id in ancestors:
                raise ValueError(f"cycle detected: {node_id} is its own ancestor")

            depth = get_depth(node_id, parent_map)

            node = DAGNodeSnapshot(
                node_id=node_id,
                run_id=run_id,
                branch_id=branch_id,
                parent_node_ids=parent_ids,
                depth=depth,
                node_metrics=node_metrics or NodeMetrics(),
            )
            self._state_store.write_dag_node(node)

            # Create PARENT edges
            for pid in parent_ids:
                edge = DAGEdgeSnapshot(
                    source_node_id=pid,
                    target_node_id=node_id,
                    edge_type=EdgeType.PARENT,
                    weight=1.0,
                )
                self._state_store.write_dag_edge(edge)

            return node

        def get_node(self, node_id: str) -> DAGNodeSnapshot | None:
            return self._state_store.load_dag_node(node_id)

        def list_nodes(self, run_id: str) -> list[DAGNodeSnapshot]:
            return self._state_store.list_dag_nodes(run_id)

        def get_ancestors(self, node_id: str, run_id: str) -> set[str]:
            parent_map = self._build_parent_map(run_id)
            return get_ancestors(node_id, parent_map)

        def get_descendants(self, node_id: str, run_id: str) -> set[str]:
            child_map = self._build_child_map(run_id)
            return get_descendants(node_id, child_map)

        def get_frontier(self, run_id: str) -> set[str]:
            parent_map = self._build_parent_map(run_id)
            all_ids = set(parent_map.keys())
            return get_frontier(parent_map, all_ids)

        def update_node_metrics(self, node_id: str, metrics: NodeMetrics) -> DAGNodeSnapshot:
            node = self._state_store.load_dag_node(node_id)
            if node is None:
                raise KeyError(f"node not found: {node_id}")
            updated = node.model_copy(update={"node_metrics": metrics})
            self._state_store.write_dag_node(updated)
            return updated

        def _build_parent_map(self, run_id: str) -> dict[str, list[str]]:
            nodes = self._state_store.list_dag_nodes(run_id)
            return {n.node_id: list(n.parent_node_ids) for n in nodes}

        def _build_child_map(self, run_id: str) -> dict[str, list[str]]:
            nodes = self._state_store.list_dag_nodes(run_id)
            child_map: dict[str, list[str]] = {n.node_id: [] for n in nodes}
            for n in nodes:
                for pid in n.parent_node_ids:
                    if pid in child_map:
                        child_map[pid].append(n.node_id)
            return child_map
    ```

       ```python
       def compute_generalization_signals(
           *,
           validation_score: float,
           training_score: float | None = None,
           cross_fold_variance: float | None = None,
           score_trend: float | None = None,
       ) -> tuple[float, float]:
           """Compute generalization_gap and overfitting_risk from stage results.

           Returns:
               (generalization_gap, overfitting_risk)
               generalization_gap: validation vs training score difference. Higher = worse.
               overfitting_risk: composite [0,1]. Higher = more risk.
           """
           if training_score is None:
               return (0.0, 0.0)

           gap = max(0.0, training_score - validation_score)

           # Overfitting risk: composite of gap magnitude, cross-fold variance, score trend
           risk_components: list[float] = [min(1.0, gap * 2.0)]  # gap contribution
           if cross_fold_variance is not None:
               risk_components.append(min(1.0, cross_fold_variance * 5.0))  # variance contribution
           if score_trend is not None and score_trend < 0:
               risk_components.append(min(1.0, abs(score_trend) * 3.0))  # declining trend contribution

           overfitting_risk = min(1.0, sum(risk_components) / len(risk_components))
           return (round(gap, 4), round(overfitting_risk, 4))
       ```

       ```python
       from __future__ import annotations
       from dataclasses import dataclass
       from v3.algorithms.decay import cosine_decay
       from v3.contracts.exploration import DAGNodeSnapshot, NodeMetrics
       from v3.orchestration.dag_service import DAGService
       from v3.ports.state_store import StateStorePort

       @dataclass(frozen=True)
       class ParentRecommendation:
           parent_node_ids: list[str]
           rationale: str
           budget_ratio: float
           diversity_weight: float

       class SelectParentsService:
           """Answers 'which past nodes should feed context into a new iteration.'

           This is NOT 'select next branch' (SelectionService). SelectParentsService
           picks parent nodes for reasoning context, code base, and performance benchmark.
           """

           def __init__(self, state_store: StateStorePort, dag_service: DAGService) -> None:
               self._state_store = state_store
               self._dag_service = dag_service

           def select_parents(
               self,
               *,
               run_id: str,
               branch_id: str,
               max_parents: int | None = None,
           ) -> ParentRecommendation:
               run = self._state_store.load_run_snapshot(run_id)
               if run is None:
                   raise KeyError(f"run not found: {run_id}")

               budget_ratio = run.current_round / max(run.max_rounds, 1)
               diversity_weight = cosine_decay(budget_ratio)

               # Determine K based on budget phase
               if max_parents is not None:
                   k = max_parents
               elif budget_ratio < 0.3:
                   k = 3  # Early: multiple parents for diversity
               else:
                   k = 1  # Iteration/Merge: single best parent (Phase 26 fallback)

               # Get all DAG nodes for this branch
               all_nodes = self._dag_service.list_nodes(run_id)
               branch_nodes = [n for n in all_nodes if n.branch_id == branch_id]
               if not branch_nodes:
                   raise KeyError(f"no DAG nodes for branch {branch_id} in run {run_id}")

               # Score candidates using 3D signal model
               scored = self._score_candidates(branch_nodes, diversity_weight)

               # Select top-K
               scored.sort(key=lambda x: x[1], reverse=True)
               selected = [node_id for node_id, _ in scored[:k]]

               rationale = (
                   f"Selected {len(selected)} parent(s) at budget_ratio={budget_ratio:.2f} "
                   f"(diversity_weight={diversity_weight:.2f}). "
                   f"Phase: {'early' if budget_ratio < 0.3 else 'iteration' if budget_ratio < 0.8 else 'merge'}."
               )

               return ParentRecommendation(
                   parent_node_ids=selected,
                   rationale=rationale,
                   budget_ratio=budget_ratio,
                   diversity_weight=diversity_weight,
               )

           def _score_candidates(
               self,
               nodes: list[DAGNodeSnapshot],
               diversity_weight: float,
           ) -> list[tuple[str, float]]:
               """Score nodes by quality (1 - diversity_weight) + diversity (diversity_weight)."""
               scored: list[tuple[str, float]] = []
               for node in nodes:
                   m = node.node_metrics
                   quality = m.validation_score * (1.0 - m.overfitting_risk) * (1.0 - min(1.0, m.generalization_gap))
                   diversity = m.diversity_score
                   combined = quality * (1.0 - diversity_weight) + diversity * diversity_weight
                   scored.append((node.node_id, combined))
               return scored
       ```
