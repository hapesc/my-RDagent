               from collections import Counter
               from v3.algorithms.decay import category_entropy
               cat_counts = Counter(spec.approach_category.value for spec in request.hypothesis_specs)
               diversity = category_entropy(dict(cat_counts))

                   branch_id=branch_id,
                   parent_node_ids=[],  # Root nodes for first layer

               )
               dag_node_ids.append(node.node_id)
