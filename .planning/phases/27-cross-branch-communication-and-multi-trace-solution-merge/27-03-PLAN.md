    ```python
    def _score_candidates_with_complementarity(
        self,
        nodes: list[DAGNodeSnapshot],
        diversity_weight: float,
        first_parent_node: DAGNodeSnapshot,
    ) -> list[tuple[str, float]]:
        """Score second parent candidates by complementarity to the first parent.

        Uses both ComponentClass coverage distance and semantic distance (1 - S_ij)
        as required by the locked decision in CONTEXT.md.
        """
        from v3.algorithms.complementarity import (
            combined_complementarity,
            component_coverage_distance,
            semantic_complementarity,
        )

        first_comp_score = first_parent_node.node_metrics.complementarity_score
        scored: list[tuple[str, float]] = []
        for node in nodes:
            if node.node_id == first_parent_node.node_id:
                continue
            metrics = node.node_metrics
            quality = (
                metrics.validation_score
                * (1.0 - metrics.overfitting_risk)
                * (1.0 - min(1.0, metrics.generalization_gap))
            )
            # Compute pairwise complementarity: coverage + semantic distance
            # Coverage distance: difference in complementarity_score as proxy
            coverage_dist = abs(metrics.complementarity_score - first_comp_score)
            # Semantic distance: 1 - diversity overlap (higher diversity_score diff = more complementary)
            semantic_dist = abs(metrics.diversity_score - first_parent_node.node_metrics.diversity_score)
            comp = combined_complementarity(coverage_dist, semantic_dist)

            combined = (
                quality * 0.4
                + comp * 0.6  # Complementarity dominates in merge stage
            )
            scored.append((node.node_id, combined))
        return scored
    ```
