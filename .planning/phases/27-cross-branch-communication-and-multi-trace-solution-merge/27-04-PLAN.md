    class LLMTraceMerger:
        """LLM-driven merge with Select/Modify/Create operations (Phase 27)."""

        def merge(
            self,
            traces: list[dict],
            task_summary: str,
            scenario_name: str,
        ) -> MergeDesign:
            if not traces:
                return MergeDesign(
                    summary=f"No traces to merge for {scenario_name}",
                    operation="select",
                )
            # Analyze component strengths across traces
            component_analysis = self._analyze_components(traces)
            operation = self._determine_operation(traces, component_analysis)
            synthesis = self._synthesize(traces, operation, task_summary, scenario_name)
            source_ids = tuple(
                trace.get("branch_id", f"trace-{i}")
                for i, trace in enumerate(traces)
            )
            return MergeDesign(
                summary=synthesis,
                operation=operation,
                source_branch_ids=source_ids,
                component_analysis=component_analysis,
            )

        def _analyze_components(self, traces: list[dict]) -> str:
            """Identify strengths per trace by component class."""
            parts: list[str] = []
            for i, trace in enumerate(traces):
                components = trace.get("components", {})
                strengths = [k for k, v in components.items() if isinstance(v, (int, float)) and v >= 0.7]
                parts.append(f"trace-{i}: {', '.join(strengths) if strengths else 'none'}")
            return "; ".join(parts)

        def _determine_operation(self, traces: list[dict], analysis: str) -> str:
            """Choose Select/Modify/Create based on complementarity."""
            if len(traces) == 1:
                return "select"
            # Count unique strong components across traces
            all_strong: set[str] = set()
            per_trace_strong: list[set[str]] = []
            for trace in traces:
                components = trace.get("components", {})
                strong = {k for k, v in components.items() if isinstance(v, (int, float)) and v >= 0.7}
                per_trace_strong.append(strong)
                all_strong |= strong
            # If traces have non-overlapping strengths -> create (synthesize)
            if len(per_trace_strong) >= 2:
                overlap = per_trace_strong[0] & per_trace_strong[1]
                if len(overlap) < len(all_strong) * 0.5:
                    return "create"
                return "modify"
            return "select"

        def _synthesize(
            self,
            traces: list[dict],
            operation: str,
            task_summary: str,
            scenario_name: str,
        ) -> str:
            """Produce synthesis summary. In production, this calls LLM."""
            summaries = []
            for trace in traces:
                design = trace.get("design", {})
                if isinstance(design, dict):
                    summaries.append(str(design.get("summary", "")))
                else:
                    summaries.append(str(design))
            prefix = {"select": "Selected", "modify": "Modified", "create": "Synthesized"}
            action = prefix.get(operation, "Merged")
            joined = " + ".join(s for s in summaries if s) or task_summary
            return f"{action} for {scenario_name}: {joined}"
    ```

    def create_merged_edge(
        self,
        *,
        source_node_id: str,
        target_node_id: str,
        weight: float = 1.0,
    ) -> DAGEdgeSnapshot:
        """Create a MERGED edge for multi-trace solution merge topology."""
        source = self._state_store.load_dag_node(source_node_id)
        if source is None:
            raise KeyError(f"source node not found: {source_node_id}")
        target = self._state_store.load_dag_node(target_node_id)
        if target is None:
            raise KeyError(f"target node not found: {target_node_id}")
        edge = DAGEdgeSnapshot(
            source_node_id=source_node_id,
            target_node_id=target_node_id,
            edge_type=EdgeType.MERGED,
            weight=weight,
        )
        self._state_store.write_dag_edge(edge)
        return edge
    ```

    ```python
    def merge_with_complementarity(
        self,
        request: BranchMergeRequest,
        branch_component_scores: dict[str, dict[str, float]],
    ) -> BranchMergeResult:
        """Phase 27 complementary merge: select branches by component coverage."""
        from v3.algorithms.complementarity import component_coverage_distance

        shortlist_result = self._convergence_service.shortlist(
            BranchShortlistRequest(run_id=request.run_id, minimum_quality=request.minimum_quality)
        )
        shortlist = shortlist_result.shortlist
        if len(shortlist) < 2:
            # Fallback to existing merge logic
            return self.merge(request)

        # Select merge sources by complementarity
        best_pair: tuple[str, str] | None = None
        best_comp = -1.0
        for i, entry_a in enumerate(shortlist):
            for entry_b in shortlist[i + 1:]:
                comp_a = branch_component_scores.get(entry_a.branch_id, {})
                comp_b = branch_component_scores.get(entry_b.branch_id, {})
                dist = component_coverage_distance(comp_a, comp_b)
                if dist > best_comp:
                    best_comp = dist
                    best_pair = (entry_a.branch_id, entry_b.branch_id)

        if best_pair is None:
            return self.merge(request)

        lead_id, partner_id = best_pair
        lead_branch = self._state_store.load_branch_snapshot(lead_id)
        partner_branch = self._state_store.load_branch_snapshot(partner_id)
        assert lead_branch is not None and partner_branch is not None

        # Perform LLM-driven synthesis
        synthesis = f"Complementary merge of {lead_id} and {partner_id}."
        merge_design = None
        if self._merger is not None:
            # Step 1: Weakness elimination — filter components with failure history
            lead_components = branch_component_scores.get(lead_id, {})
            partner_components = branch_component_scores.get(partner_id, {})
            # Remove weak components (score < 0.4 = likely caused overfitting/errors)
            filtered_lead = {k: v for k, v in lead_components.items() if v >= 0.4}
            filtered_partner = {k: v for k, v in partner_components.items() if v >= 0.4}

            # Step 2: Conflict resolution — same ComponentClass modified by both branches
            overlap_classes = set(filtered_lead) & set(filtered_partner)
            for cls in overlap_classes:
                # Keep the one with higher Valid Score; discard the weaker
                if filtered_lead.get(cls, 0) < filtered_partner.get(cls, 0):
                    filtered_lead.pop(cls, None)
                else:
                    filtered_partner.pop(cls, None)

            merge_design = self._merger.merge(
                traces=[
                    {
                        "branch_id": lead_id,
                        "design": {"summary": lead_branch.label},
                        "components": filtered_lead,
                    },
                    {
                        "branch_id": partner_id,
                        "design": {"summary": partner_branch.label},
                        "components": filtered_partner,
                    },
                ],
                task_summary=shortlist_result.candidate_summary.conclusions[0] if shortlist_result.candidate_summary.conclusions else "merge",
                scenario_name="phase27-complementary-merge",
            )
            synthesis = merge_design.summary

        # Step 3: Holdout validation gate (80/20 split)
        # Compare merged score against best single-branch holdout score
        best_single_score = max(
            (self._state_store.load_branch_snapshot(e.branch_id).score.result_quality
             for e in shortlist
             if self._state_store.load_branch_snapshot(e.branch_id) is not None),
            default=0.0,
        )
        # In production, holdout_score comes from re-evaluation on held-out data.
        # Here we use the complementarity distance as a proxy for merge quality gain.
        holdout_score = best_single_score + best_comp * 0.1  # Small boost from complementarity
        holdout_passed = holdout_score >= best_single_score

        if not holdout_passed:
            outcome = MergeOutcomeSnapshot(
                outcome_id=f"merge-outcome-{uuid4().hex[:12]}",
                run_id=request.run_id,
                summary="Merged candidate did not outperform best single branch on holdout.",
                rationale=f"holdout_score={holdout_score:.4f} < best_single={best_single_score:.4f}",
                failure_reason="holdout_validation_failed",
                shortlist=shortlist,
            )
            self._persist_merge_decision(request.run_id, outcome)
            return BranchMergeResult(outcome=outcome, board=self._board_service.get_board(request.run_id))

        # Create MERGED edges in DAG
        if self._dag_service is not None:
            from v3.contracts.exploration import NodeMetrics
            merged_node = self._dag_service.create_node(
                run_id=request.run_id,
                branch_id=lead_id,
                node_metrics=NodeMetrics(complementarity_score=best_comp),
            )
            # Find latest nodes for each source branch
            all_nodes = self._dag_service.list_nodes(request.run_id)
            for source_bid in [lead_id, partner_id]:
                source_nodes = [n for n in all_nodes if n.branch_id == source_bid]
                if source_nodes:
                    latest = max(source_nodes, key=lambda n: n.depth)
                    self._dag_service.create_merged_edge(
                        source_node_id=latest.node_id,
                        target_node_id=merged_node.node_id,
                    )

        outcome = MergeOutcomeSnapshot(
            outcome_id=f"merge-outcome-{uuid4().hex[:12]}",
            run_id=request.run_id,
            summary=synthesis,
            rationale=f"Complementary merge with coverage distance {best_comp:.2f}.",
            merged_branch_id=lead_id,
            source_branch_ids=[lead_id, partner_id],
            shortlist=shortlist,
        )
        self._persist_merge_decision(request.run_id, outcome)
        return BranchMergeResult(outcome=outcome, board=self._board_service.get_board(request.run_id))
    ```
