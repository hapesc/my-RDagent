    ```python
    # Signal 4: Collect ComponentClass data for functional preservation (Phase 27)
    branch_component_classes: dict[str, set[str]] | None = None
    global_best_component_classes: set[str] | None = None

    if use_multi_signal:
        # Extract component classes from branch HypothesisSpec if available
        comp_data: dict[str, set[str]] = {}
        for branch in active_branches:
            hypothesis = self._state_store.load_hypothesis_spec(branch.branch_id)
            if hypothesis is not None and hypothesis.component_classes:
                comp_data[branch.branch_id] = set(hypothesis.component_classes)
        if comp_data:
            branch_component_classes = comp_data
            # Global best = highest scoring branch's components
            best_branch = max(active_branches, key=lambda b: b.score.result_quality)
            global_best_component_classes = comp_data.get(best_branch.branch_id, set())
    ```

    ```python
    class TestPruneSignal4Integration:

        def test_unique_component_branch_survives(self):
    ```
