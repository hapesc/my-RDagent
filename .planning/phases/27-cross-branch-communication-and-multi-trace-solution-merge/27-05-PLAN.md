    """Integration tests for Phase 27 cross-branch communication and merge."""

    from __future__ import annotations

    import pytest
    from unittest.mock import MagicMock, patch

        def _make_dag_node(self, node_id: str, branch_id: str, run_id: str, score: float, depth: int = 1) -> DAGNodeSnapshot:
            return DAGNodeSnapshot(
                node_id=node_id, run_id=run_id, branch_id=branch_id,
                depth=depth, node_metrics=NodeMetrics(validation_score=score),
            )

        def test_global_best_identified_from_frontier(self):
            # Setup mock state store + DAG with 3 frontier nodes
            from v3.orchestration.dag_service import DAGService
            from v3.orchestration.branch_share_service import BranchShareService
            state = MagicMock()
            dag = MagicMock(spec=DAGService)
            nodes = [
                self._make_dag_node("n1", "b1", "run1", 0.5),
                self._make_dag_node("n2", "b2", "run1", 0.9),
                self._make_dag_node("n3", "b3", "run1", 0.3),
            ]
            dag.get_frontier.return_value = {"n1", "n2", "n3"}
            dag.get_node.side_effect = lambda nid: next((n for n in nodes if n.node_id == nid), None)
            service = BranchShareService(state, MagicMock(), dag_service=dag, embedding_port=StubEmbeddingPort())
            assert service.identify_global_best("run1") == "b2"

        def test_round_0_no_sharing(self):
            from v3.orchestration.branch_share_service import BranchShareService
            state = MagicMock()
            service = BranchShareService(state, MagicMock(), dag_service=MagicMock(), embedding_port=StubEmbeddingPort())
            assert service.compute_sharing_candidates(
                run_id="run1", target_branch_id="b1", current_round=0, budget_ratio=0.0
            ) == []


    class TestPruneSignal4Integration:
        """Signal 4 functional preservation in pruning pipeline."""

        def test_unique_component_branch_survives(self):
            candidates = [("best", 0.9), ("unique_comp", 0.3), ("weak", 0.2)]
            pruned = prune_branch_candidates(
                candidates,
                relative_threshold=0.5,
                branch_component_classes={
                    "best": {"model"},
                    "unique_comp": {"model", "ensemble"},  # ensemble is unique
                    "weak": {"model"},
                },
                global_best_component_classes={"model"},
                min_active_branches=1,
            )
            # unique_comp has ensemble not in global best -> exempt
            assert "unique_comp" not in pruned
            assert "weak" in pruned

        def test_no_unique_component_branch_pruned(self):
            candidates = [("best", 0.9), ("dup", 0.2)]
            pruned = prune_branch_candidates(
                candidates,
                relative_threshold=0.5,
                branch_component_classes={"best": {"model"}, "dup": {"model"}},
                global_best_component_classes={"model"},
                min_active_branches=1,
            )
            assert "dup" in pruned


    class TestComplementaryMergeIntegration:
        """Complementary merge: component analysis → LLM synthesis → MERGED edges."""

        def test_complementary_pair_selected(self):
            from v3.algorithms.complementarity import component_coverage_distance
            a = {"model": 0.9, "feature_eng": 0.2}
            b = {"model": 0.2, "feature_eng": 0.9}
            c = {"model": 0.8, "feature_eng": 0.1}
            # a-b should have highest coverage distance
            assert component_coverage_distance(a, b) > component_coverage_distance(a, c)
            assert component_coverage_distance(a, b) > component_coverage_distance(b, c)


    class TestFullLifecycleIntegration:
        """Share → Prune → Merge in sequence (conceptual flow)."""

        def test_full_round_lifecycle(self):
            # Phase 27 adds: sharing -> pruning with signal 4 -> merge

            # pruning preserves unique components, merge uses complementarity

                relative_threshold=0.5,
                branch_component_classes={
                    "a": {"model", "feature_eng"},
                    "b": {"model", "ensemble"},
                    "c": {"workflow"},  # unique -> preserved
                    "d": {"model"},
                },
                global_best_component_classes={"model", "feature_eng"},
                min_active_branches=2,
            )
            assert "c" not in pruned  # unique workflow component preserved
            assert "d" in pruned  # no unique components

            # Step 2: Complementarity check for merge candidates
            from v3.algorithms.complementarity import component_coverage_distance
            comp_ab = component_coverage_distance(
                {"model": 0.9, "feature_eng": 0.9},
                {"model": 0.5, "ensemble": 0.8},
            )
            assert comp_ab > 0  # complementary pair identified
