import random
from v3.ports.holdout_port import FoldSpec

class DefaultHoldoutSplitPort:
    """Production K-fold splitter with real random shuffle."""

    def __init__(self, k: int = 5, seed: int | None = None) -> None:
        self._k = k
        self._seed = seed

    def split(self, *, run_id: str) -> list[FoldSpec]:
        rng = random.Random(self._seed)
        indices = list(range(self._k))
        rng.shuffle(indices)
        return [
            FoldSpec(
                fold_index=idx,
                train_ref=f"{run_id}-default-train-{idx}",
                holdout_ref=f"{run_id}-default-holdout-{idx}",
            )
            for idx in indices
        ]
```

import math
from collections import Counter

class DefaultEmbeddingPort:
    """Lightweight TF-IDF embedding using only Python stdlib."""

    def __init__(self, dim: int = 256) -> None:
        self._dim = dim

    def embed(self, texts: list[str]) -> list[list[float]]:
        if not texts:
            return []
        # Build vocabulary from all texts
        all_tokens: list[list[str]] = [self._tokenize(t) for t in texts]
        vocab: dict[str, int] = {}
        doc_freq: Counter[str] = Counter()
        for tokens in all_tokens:
            unique = set(tokens)
            for token in unique:
                doc_freq[token] += 1
            for token in tokens:
                if token not in vocab:
                    vocab[token] = len(vocab)
        n_docs = len(texts)
        # Compute TF-IDF vectors, hashed to fixed dim
        vectors: list[list[float]] = []
        for tokens in all_tokens:
            vec = [0.0] * self._dim
            tf = Counter(tokens)
            for token, count in tf.items():
                tf_val = count / max(len(tokens), 1)
                idf_val = math.log((1 + n_docs) / (1 + doc_freq.get(token, 0))) + 1.0
                bucket = hash(token) % self._dim
                vec[bucket] += tf_val * idf_val
            # L2 normalize
            norm = math.sqrt(sum(v * v for v in vec))
            if norm > 0:
                vec = [v / norm for v in vec]
            vectors.append(vec)
        return vectors

    @staticmethod
    def _tokenize(text: str) -> list[str]:
        return text.lower().split()
```
